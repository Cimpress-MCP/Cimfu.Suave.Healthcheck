#r "Suave.dll"
#r "Cimfu.Suave.Healthcheck.dll"
open Suave
open Suave.Types
open Cimfu.Suave.Healthcheck

let myApp : WebPart = Http.RequestErrors.NOT_FOUND "Not here"

let healthchecks = Map.ofList ["noop", Checks.noop]

let app =
  myApp
  |> withHealthcheck healthchecks
